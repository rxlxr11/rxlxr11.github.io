<template>

    <div id="map-container" style="width: 100%; height: 80vh;"></div>
    <button @click="hideLayer">隐藏图层</button>
    <button @click="showLayer">展示图层</button>

    <button @click="addLayer">添加图层</button>
    <button @click="removeLayer">移除图层</button>

</template>

<script setup>
import AMapLoader from '@amap/amap-jsapi-loader';
import { onMounted, onUnmounted } from 'vue';

let map = null;
let traffic = null;

const showLayer = () => {
    if (map.hasLayer(traffic)) {
        console.warn("Traffic layer is already visible.");
        return;
    }

    traffic.show(); // 显示图层

}

const hideLayer = () => {
    if (!map.hasLayer(traffic)) {
        console.warn("Traffic layer is not visible.");
        return;
    }
    traffic.hide(); // 隐藏图层
}

const addLayer = () => {

    if (map.hasLayer(traffic)) {
        console.warn("Traffic layer is already added.");
        return;
    }

    map.add(traffic); // 通过add方法添加图层
}

const removeLayer = () => {

    if (!map.hasLayer(traffic)) {
        console.warn("Traffic layer is not added.");
        return;
    }
    map.remove(traffic); // 通过remove方法移除图层
}   


onMounted(() => {
    AMapLoader.load({
        key: "", // 申请好的Web端开发者Key，首次调用 load 时必填
        version: "2.0", // 指定要加载的 JSAPI 的版本，缺省时默认为 1.4.15
        plugins: ["AMap.Scale"], //需要使用的的插件列表，如比例尺'AMap.Scale'，支持添加多个如：['...','...']
    })
        .then((AMap) => {
            const defaultLayer = new AMap.createDefaultLayer({
                zooms: [3, 20], //可见级别
                visible: true, //是否可见
                opacity: 1, //透明度
                zIndex: 0, //叠加层级
            });

            const roadNetLayer = new AMap.TileLayer.RoadNet({
                zIndex: 1
            });


            map = new AMap.Map("map-container", {
                // 设置地图容器id
                viewMode: "2D", // 是否为3D地图模式
                zoom: 11, // 初始化地图级别
                center: [116.397428, 39.90923], // 初始化地图中心点位置
                layers: [defaultLayer, roadNetLayer], //layer为创建的默认图层
            });

            traffic = new AMap.TileLayer.Traffic({
                autoRefresh: true, //是否自动刷新，默认为false
                interval: 180, //刷新间隔，默认180s
            });

            map.add(traffic); //通过add方法添加图层
        })
        .catch((e) => {
            console.log(e);
        });
});

onUnmounted(() => {
  map?.destroy();
});

</script>

<template>

    <div id="map-container" style="width: 100%; height: 80vh;"></div>

    <div class='input-card'>
        <div class="input-item">
            <input type="checkbox" @click="toggleScale" />比例尺
        </div>

        <div class="input-item">
            <input type="checkbox" id="toolbar" @click="toggleToolBar" />工具条
        </div>

        <div class="input-item">
            <input type="checkbox" id="controlBar" @click="toggleControlBar" />工具条方向盘
        </div>

        <div class="input-item">
            <input type="checkbox" id="overview" @click="toggleOverViewShow" />显示鹰眼
        </div>
    </div>

</template>

<script setup>
import AMapLoader from '@amap/amap-jsapi-loader';
import { onMounted, onUnmounted } from 'vue';

let map = null;

let overView = null;
let toolbar = null;
let controlBar = null;
let scale = null;

const toggleOverViewShow = (event) => {
    var checked = event.target.checked;
    if (checked) {
        overView.show();
    } else {
        overView.hide();
    }
}

const toggleControlBar = (event) => {
    var checked = event.target.checked;
    if (checked) {
        controlBar.show();
    } else {
        controlBar.hide();
    }
}

const toggleScale = (event) => {
    var checked = event.target.checked;
    if (checked) {
        scale.show();
    } else {
        scale.hide();
    }
}

const toggleToolBar = (event) => {
    var checked = event.target.checked;
    if (checked) {
        toolbar.show();
    } else {
        toolbar.hide();
    }
}


onMounted(() => {
    AMapLoader.load({
        key: "", // 申请好的Web端开发者Key，首次调用 load 时必填
        version: "2.0", // 指定要加载的 JSAPI 的版本，缺省时默认为 1.4.15
        plugins: ['AMap.ToolBar', 'AMap.ControlBar', 'AMap.Scale', 'AMap.HawkEye'] //加载插件,
    })
        .then((AMap) => {

            map = new AMap.Map("map-container", {
                // 设置地图容器id
                viewMode: "2D", // 是否为3D地图模式
                zoom: 11, // 初始化地图级别
                center: [116.397428, 39.90923], // 初始化地图中心点位置
                
            });

            toolbar = new AMap.ToolBar();
            controlBar = new AMap.ControlBar();
            scale = new AMap.Scale();
            overView = new AMap.HawkEye();

            map.addControl(toolbar);
            map.addControl(controlBar);
            map.addControl(scale);
            map.addControl(overView);
        })
        .catch((e) => {
            console.log(e);
        });
});

onUnmounted(() => {
    map?.destroy();
});


</script>

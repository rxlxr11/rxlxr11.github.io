import AMapLoader from '@amap/amap-jsapi-loader';

const KEY = "......";
export const DEFAULT_MAP_ZOOM = 14;
export const DEFAULT_MAP_CENTER = [118.172505, 24.679562];

export async function initMap(containerId, center = DEFAULT_MAP_CENTER, zoom = DEFAULT_MAP_ZOOM) {

    return new Promise((resolve, reject) => {
        AMapLoader.load({
            key: KEY,
            version: "2.0",
            plugins: ["AMap.Marker", "AMap.InfoWindow"]
        })
        .then((AMap) => {
            const map = new AMap.Map(containerId, {
                viewMode: '2D',
                zoom: zoom,
                center: center,
            });
            resolve(map);
        })
        .catch((error) => {
            console.error('åœ°å›¾åŠ è½½é”™è¯¯', error);
            reject(error);
        });
    });

}
